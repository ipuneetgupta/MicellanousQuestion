#include <iostream>
using namespace std;
int main() {
    int t;
    cin>>t;
    while(t--){
        int x,y,z;
        cin>>x>>y>>z;
        if(x+y-z==0){
             cout<<"yes"<<endl;
        }else if(x+z-y==0){
             cout<<"yes"<<endl;

        }else if(y+z-x==0){
             cout<<"yes"<<endl;

        }else{
             cout<<"no"<<endl;

        }
    }
}
